scaleSpelling.js

This module contains utility functions to create scale and chord structures useable with Tone.js

scaleSpelling.js
Example
use the script tag to add this file
<script src="path/to/scaleSpelling.js"></script>

TransposeSequence

this function transposes an arrayOfNotes by a specified number of half steps

new TransposeSequence(arrayOfNotes: object, transposeHalfSteps: number): object
Parameters
arrayOfNotes (object)
transposeHalfSteps (number)
Returns
object: array of transposed notes NOTE: doesn't deal with correct name but only the correct sounding pitch
Example
var c_scale = ["C4","D4","E4","F4","G4","A4","B4","C5"];
var eb_scale = TransposeSequence(c_scale, 3);
// eb_scale = ["D#4","F4","G4","G#4","A#4","C5","D5","D$5"];
// wrong spelling but sounds correct.  
// This spelling is OK for (guitar/mandolin) tablature but can't be used for standard notation.

extendScaleToThreeOctaves

this function takes a scale and extends it to three octave (one octave lower and one octave higher)

extendScaleToThreeOctaves(scale: object): object
Parameters
scale (object) (array of strings)
Returns
object:
Example
var my_scale = ["A4","B4","C#5","D5","E5","F#5","G#5","A5"];
var new_scale = extendScaleToThreeOctaves(my_scale);
// new_scale = ["A3","B3","C#4","D4","E4","F#4","G#4",
//               "A4","B4","C#5","D5","E5","F#5","G#5",
//                "A5","B5","C#6","D6","E6","F#6","G#6","A6"]

spellMode

this function returns an array of noteNames suitable for Tone.js use

spellMode(scaleFormulaInput: (string | object), rootAndOctave: string, modeNum: number): object
Parameters
scaleFormulaInput ((string | object)) (can be either a (comma separated) string or an array of integers)
rootAndOctave (string)
modeNum (number)
Returns
object: array of noteNames in mode
Example
const myFormula = "0,2,4,5,7,9,11";
var myMode = spellMode(myFormula, "Eb4",3); // third mode of G major
// myMode = ['G4','Ab4',Bb4','C5','D5','Eb5','F5','G5'] (G phrygian)

spellDiatonicChord

this function returns an array of noteNames in chord suitable for Tone.js use

spellDiatonicChord(scaleDegrees: string, scaleFormula: string, rootAndOctave: string): string
Parameters
scaleDegrees (string) (can be either a (comma separated) string or an array of strings)
scaleFormula (string)
rootAndOctave (string)
Returns
string: array of noteNames in chord
Example
var scale_degrees = "2,4,6"; // or [2,4,6]
var scale_formula = "0,2,4,5,7,9,11" // or [0,2,4,5,7,9,11]
var myChord = spellDiatonicChord(scale_degrees, scale_formula, "A4")
// myChord = ["B4","D5","F#5"]

spellChord

this function returns an array of noteNames suitable for Tone.js use

spellChord(chordFormulaInput: (string | object), rootAndOctave: string, scaleDegrees: object): object
Parameters
chordFormulaInput ((string | object)) (can be either a (comma separated) string or an array of integers)
rootAndOctave (string)
scaleDegrees (object) (an array of integers)
Returns
object: array of noteNames in chord
Example
var chord_formula = "0,4,7";
var root = "A4";
var scale_degrees = [1,3,5];
var myChord = spellChord(chord_formula, root, scale_degrees);
// myChord = ["A4","C#5","E5"]

// myInterval = ["Bb4","F5"]

spellScale

spellScale(scaleFormulaInput: (string | object), rootAndOctave: string): object
Parameters
scaleFormulaInput ((string | object)) (can be either a (comma separated) string or an array of integers)
rootAndOctave (string)
Returns
